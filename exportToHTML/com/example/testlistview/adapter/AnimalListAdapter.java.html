24   public class AnimalListAdapter extends ArrayAdapter<Animal> { 
25    
26       private Context mContext; 
27       private ArrayList<Animal> mAnimalList; 
28       private int mLayoutResId; 
29    
30       public AnimalListAdapter(@NonNull Context context, @LayoutRes int resource, @NonNull ArrayList<Animal> objects) { 
31           super(context, resource, objects); 
32    
33           this.mContext = context; 
34           this.mLayoutResId = resource; 
35           this.mAnimalList = objects; 
36       } 
37    
38       @NonNull 
39       @Override 
40       public View getView(int position, @Nullable View convertView, @NonNull ViewGroup parent) { 
41           LayoutInflater inflater = LayoutInflater.from(mContext); 
42           View v = inflater.inflate(mLayoutResId, null); 
43    
44           ImageView iv = v.findViewById(R.id.imageView); 
45           TextView tv = v.findViewById(R.id.textView); 
46    
47           Animal animal = mAnimalList.get(position); 
48    
49           iv.setImageResource(animal.picture); 
50           tv.setText( 
51                   String.format( 
52                           Locale.getDefault(), 
53                           "%s (%s)", 
54                           animal.thaiName, 
55                           animal.englishName 
56                   ) 
57           ); 
58           return v; 
59       }
